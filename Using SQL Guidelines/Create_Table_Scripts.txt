CREATE TABLE encounters(
   ID                VARCHAR(50) NOT NULL PRIMARY KEY
  ,START          	 DATE NOT NULL
  ,STOP           	 DATE NULL
  ,PATIENT           VARCHAR(50) NOT NULL
  ,PROVIDER          VARCHAR(50) NULL
  ,ENCOUNTERCLASS    VARCHAR(50) NULL
  ,CODE              INTEGER  NULL
  ,DESCRIPTION       VARCHAR(200) NULL
  ,COST              NUMERIC(6,2) NULL
  ,REASONCODE        VARCHAR(50) NULL
  ,REASONDESCRIPTION VARCHAR(200) NULL
);

CREATE TABLE patients(
   Id         VARCHAR(50) NOT NULL PRIMARY KEY
  ,BIRTHDATE  DATE  NOT NULL
  ,DEATHDATE  DATE  NULL
  ,SSN        VARCHAR(50) NULL
  ,DRIVERS    VARCHAR(50) NULL
  ,PASSPORT   VARCHAR(50) NULL
  ,PREFIX     VARCHAR(50) NULL
  ,FIRST      VARCHAR(50) NULL
  ,LAST       VARCHAR(50) NULL
  ,SUFFIX     VARCHAR(50) NULL
  ,MAIDEN     VARCHAR(50) NULL
  ,MARITAL    VARCHAR(50) NULL
  ,RACE       VARCHAR(50) NULL
  ,ETHNICITY  VARCHAR(50) NULL
  ,GENDER     VARCHAR(50) NULL
  ,BIRTHPLACE VARCHAR(200) NULL
  ,ADDRESS    VARCHAR(50) NULL
  ,CITY       VARCHAR(50) NULL
  ,STATE      VARCHAR(50) NULL
  ,ZIP        VARCHAR(12) NULL
);

CREATE TABLE allergies(
   START       DATE  NOT NULL
  ,STOP        DATE  NULL
  ,PATIENT     VARCHAR(50) NOT NULL
  ,ENCOUNTER   VARCHAR(50) NOT NULL
  ,CODE        INTEGER  NULL
  ,DESCRIPTION VARCHAR(200) NULL
);


CREATE TABLE medications(
   START             DATE  NOT NULL
  ,STOP              DATE  NULL
  ,PATIENT           VARCHAR(50) NOT NULL
  ,ENCOUNTER         VARCHAR(50) NOT NULL
  ,CODE              INTEGER  NULL
  ,DESCRIPTION       VARCHAR(200) NULL
  ,COST              NUMERIC(8,2) NULL
  ,DISPENSES         INTEGER  NULL
  ,TOTALCOST         NUMERIC(10,2) NULL
  ,REASONCODE        VARCHAR(50) NULL
  ,REASONDESCRIPTION VARCHAR(200) NULL
); 


CREATE TABLE procedures(
   DATE               DATE  NOT NULL
  ,PATIENT            VARCHAR(50) NOT NULL
  ,ENCOUNTER          VARCHAR(50) NOT NULL
  ,CODE               VARCHAR(50) NULL
  ,DESCRIPTION        VARCHAR(200) NULL
  ,COST               NUMERIC(10,2) NULL
  ,REASONCODE         VARCHAR(50) NULL
  ,REASONDESCRIPTION  VARCHAR(200) NULL
);
